class Sorvete:
    def __init__(self, sabor, preco):
        self.sabor = sabor
        self.preco = preco

    def __str__(self):
        return f"{self.sabor} - R$ {self.preco:.2f}"


class Sorveteria:
    def __init__(self):
        self.menu = [
            Sorvete("Chocolate", 5.00),
            Sorvete("Morango", 4.50),
            Sorvete("Baunilha", 4.00),
            Sorvete("Flocos", 5.50),
            Sorvete("Napolitano", 6.00)
        ]

    def exibir_menu(self):
        print("\n--- MENU DA SORVETERIA ---")
        for idx, sorvete in enumerate(self.menu, 1):
            print(f"{idx}. {sorvete}")

    def escolher_sorvete(self):
        self.exibir_menu()
        try:
            escolha = int(input("\nDigite o n√∫mero do sorvete que deseja comprar: "))
            if 1 <= escolha <= len(self.menu):
                print(f"\n‚úÖ Voc√™ escolheu {self.menu[escolha - 1].sabor}! Aproveite seu sorvete! üç¶")
            else:
                print("N√∫mero inv√°lido! Tente novamente.")
        except ValueError:
            print("Entrada inv√°lida! Digite um n√∫mero correspondente ao menu.")

    def iniciar(self):
        while True:
            print("\n--- SORVETERIA ---")
            print("1. Exibir menu")
            print("2. Escolher um sorvete")
            print("3. Sair")
            
            opcao = input("Escolha uma op√ß√£o: ")

            if opcao == "1":
                self.exibir_menu()
            elif opcao == "2":
                self.escolher_sorvete()
            elif opcao == "3":
                print("Encerrando o sistema...")
                break
            else:
                print("Op√ß√£o inv√°lida! Tente novamente.")
